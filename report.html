def visualization(n_groups, survived, not_survived):
    index = np.arange(n_groups)
    bar_width = 0.35
    opacity = 0.8
 
    plt.bar(index, survived, bar_width,
            alpha=opacity,
            color='g',
            label='Survived')
 
    plt.bar(index + bar_width, not_survived, bar_width,
            alpha=opacity,
            color='b',
            label='Not-Survived')
    
    return index+bar_width
